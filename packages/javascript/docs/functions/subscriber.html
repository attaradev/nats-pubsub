      Example

      interface UserCreatedMessage {
        id: string;
        name: string;
        email: string;
      }

      @subscriber<UserCreatedMessage>('production.app.user.created')
      class UserSubscriber {
        async call(message: UserCreatedMessage, metadata: EventMetadata) {
          // message is fully typed!
